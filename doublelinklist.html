Doubly Linked List Queue

Algorithm

class Node {
    constructor(value) {
        this.value = value;
        this.next = null;
        this.prev = null;
    }
}

class DoublyLinkedList {
    constructor() {
        this.head = null;
        this.tail = null;
    }

    enqueue(value) {
        let newNode = new Node(value);
        if (!this.head) {
            this.head = this.tail = newNode;
        } else {
            this.tail.next = newNode;
            newNode.prev = this.tail;
            this.tail = newNode;
        }
    }

    dequeue() {
        if (this.head) {
            this.head = this.head.next;
            if (this.head) this.head.prev = null;
            else this.tail = null;
        }
    }
}
